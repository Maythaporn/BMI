// Bmi.js

import React, { useState } from 'react';

function Bmi() {
  // ประกาศ state สำหรับเก็บค่าน้ำหนักและส่วนสูง
  const [weight, setWeight] = useState('');
  const [height, setHeight] = useState('');

  // ฟังก์ชันสำหรับคำนวณ BMI
  const calculateBMI = () => {
    // แปลงน้ำหนักและส่วนสูงเป็นตัวเลข
    const weightNum = parseFloat(weight);
    const heightNum = parseFloat(height) / 100; // แปลงเป็นเมตร

    // คำนวณ BMI
    if (!isNaN(weightNum) && !isNaN(heightNum) && heightNum > 0) {
      const bmi = weightNum / (heightNum * heightNum);
      return bmi.toFixed(2); // ปัดเลขทศนิยม 2 ตำแหน่ง
    } else {
      return 'Invalid Input';
    }
  };

  return (
    <div>
      <h1>BMI Calculator</h1>
      <div>
        <label>Weight (kg):</label>
        <input
          type="text"
          value={weight}
          onChange={(e) => setWeight(e.target.value)}
        />
      </div>
      <div>
        <label>Height (cm):</label>
        <input
          type="text"
          value={height}
          onChange={(e) => setHeight(e.target.value)}
        />
      </div>
      <div>
        <p>Your BMI: {calculateBMI()}</p>
      </div>
    </div>
  );
}

export default Bmi;
